        // --- SFX ENGINE ---
        const clickSfx = new Audio('Click.mp3');

        let currentSlide = 1;

        const todayDateObj = new Date();

        let isCalendarWideGrid = false;
        
        let runningStreakCount = 1;
        let hasLoggedTodayCheckIn = false;

        let historyMoodDatabase = {};
        let moodDataset = [];

        const moodImgMap = { 'Happy': 'happy.PNG', 'Cozy': 'cozy.PNG', 'Tired': 'tired.PNG', 'Awful': 'awful.PNG' };
        
        // --- UPDATED: DYNAMIC SCRIPT COLOR ASSIGNMENT ---
        const moodColorMap = {
            'Happy': 'var(--pastel-happy)',
            'Cozy': 'var(--pastel-cozy)',
            'Tired': 'var(--pastel-tired)',
            'Awful': 'var(--pastel-awful)'
        };

        const alternateRecommendations = [
            { tag: "Breathing", title: "5-Min Heart Rate Coherence" },
            { tag: "Mindset", title: "Affirmations for Steady Boundaries" },
            { tag: "Somatic", title: "Gentle Desk Stretch Release" },
            { tag: "Creative", title: "3-Minute Color Palette Doodle" },
            { tag: "Focus", title: "Ambient Lo-Fi Audio Isolation" }
        ];
        let currentRecIndex = 0;

        document.addEventListener("DOMContentLoaded", () => {
            buildCalendarTimelineDeck();
            updateStreakUI();

        function escapeHtml(text) {
            return text.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
        }

        // --- DYNAMIC TIME-BASED GREETING HANDLER ---
        function updateWelcomeGreeting() {
            const greetingEl = document.getElementById('welcome-greeting');
            if (!greetingEl) return;
            
            const hours = new Date().getHours();
            let greeting = "Hi there!";
            
            if (hours >= 5 && hours < 12) {
                greeting = "Good morning!";
            } else if (hours >= 12 && hours < 17) {
                greeting = "Good afternoon!";
            } else {
                greeting = "Good evening!";
            }
            
            greetingEl.innerHTML = `${greeting} ✨<br>Let's check in for today.`;
        }

        function handleAppLogin() {
            navigateTo('screen-home');
            updateStreakUI();
            updateWelcomeGreeting();

            setTimeout(() => {
                const streakCard = document.getElementById('streak-card');
                if (streakCard) {
                    streakCard.style.transform = 'scale(1.08) rotate(2deg)';
                    setTimeout(() => streakCard.style.transform = 'scale(1) rotate(0deg)', 220);
                }
            }, 300);
        }

        function buildCalendarTimelineDeck() {
            const scroller = document.getElementById('mini-calendar-scroller');
            if (!scroller) return;
            scroller.innerHTML = '';

            const currentMonthName = todayDateObj.toLocaleString('default', { month: 'long' });
            document.getElementById('cal-month-title').innerText = `${currentMonthName} Timeline`;

            const currentDayNumber = todayDateObj.getDate();
            const daysInMonth = new Date(todayDateObj.getFullYear(), todayDateObj.getMonth() + 1, 0).getDate();

            for (let d = 1; d <= daysInMonth; d++) {
                const dayDate = new Date(todayDateObj.getFullYear(), todayDateObj.getMonth(), d);
                const stringDayLabel = dayDate.toLocaleString('default', { weekday: 'short' });

                const node = document.createElement('div');
                node.className = 'cal-day-node';
                node.id = `cal-node-${d}`;

                if (d === currentDayNumber) node.classList.add('active-today');
                if (d === selectedCalendarDayId) node.classList.add('selected-date');

                if (d > currentDayNumber) {
                    node.classList.add('future-locked');
                    node.setAttribute('title', 'Locked Future Window');
                }

                // Inject the exact tone color onto the cell block frame if logged
                if (historyMoodDatabase[d]) {
                    node.style.backgroundColor = moodColorMap[historyMoodDatabase[d].type];
                    node.style.borderColor = 'rgba(0,0,0,0.05)';
                }

                let subDotHtml = `<div class="mood-dot-marker"></div>`;
                if (historyMoodDatabase[d]) {
                    subDotHtml = `<div class="mood-dot-marker"><img src="${historyMoodDatabase[d].img}" alt=""></div>`;
                }

                node.innerHTML = `
                    <div class="day-lbl">${stringDayLabel}</div>
                    <div class="num-lbl">${d}</div>
                    ${subDotHtml}
                `;

                if (d <= currentDayNumber) {
                    node.onclick = () => selectCalendarDateTarget(d);
                } else {
                    node.onclick = () => alert("You cannot edit or track future dates inside your workspace window! 🌸");
                }

                scroller.appendChild(node);
            }

            if (!isCalendarWideGrid) {
                setTimeout(() => {
                    const activeNode = document.getElementById(`cal-node-${currentDayNumber}`);
                    if (activeNode) {
                        scroller.scrollLeft = activeNode.offsetLeft - scroller.offsetWidth / 2 + activeNode.offsetWidth / 2;
                    }
                }, 100);
            }
        }

        function toggleCalendarWideView() {
            const scroller = document.getElementById('mini-calendar-scroller');
            const toggleBtn = document.getElementById('wide-toggle-btn');
            isCalendarWideGrid = !isCalendarWideGrid;

            if (isCalendarWideGrid) {
                scroller.classList.add('full-wide-grid');
                toggleBtn.innerText = "Row View ➔";
            } else {
                scroller.classList.remove('full-wide-grid');
                toggleBtn.innerText = "Full View 🗓️";
                buildCalendarTimelineDeck(); 
            }
        }

        function selectCalendarDateTarget(dayId) {
            selectedCalendarDayId = dayId;
            document.querySelectorAll('.cal-day-node').forEach(n => n.classList.remove('selected-date'));
            
            const selectedNode = document.getElementById(`cal-node-${dayId}`);
            if (selectedNode) selectedNode.classList.add('selected-date');

            // Reset active styles on badges
            document.querySelectorAll('.mood-badge').forEach(badge => {
                badge.classList.remove('selected', 'select-Happy', 'select-Cozy', 'select-Tired', 'select-Awful');
            });

            if (historyMoodDatabase[dayId]) {
                const loggedMoodType = historyMoodDatabase[dayId].type;
                const activeBadge = document.getElementById(`badge-${loggedMoodType}`);
                if (activeBadge) {
                    activeBadge.classList.add('selected', `select-${loggedMoodType}`);
                }
            }
        }

        function updateStreakUI() {
            const counterEl = document.getElementById('live-streak-counter');
            if (counterEl) {
                counterEl.innerText = `🔥 ${runningStreakCount}`;
            }
        }

        function rerollRecommendation() {
            const card = document.getElementById('activity-card');
            card.style.opacity = 0;
            setTimeout(() => {
                currentRecIndex = (currentRecIndex + 1) % alternateRecommendations.length;
                const nextRec = alternateRecommendations[currentRecIndex];
                document.getElementById('rec-tag').innerText = nextRec.tag;
                document.getElementById('rec-title').innerText = nextRec.title;
                card.style.opacity = 1;
            }, 250);
        }

        function logMood(badgeElement, moodName) {
            document.querySelectorAll('.mood-badge').forEach(badge => {
                badge.classList.remove('selected', 'select-Happy', 'select-Cozy', 'select-Tired', 'select-Awful');
            });

            badgeElement.classList.add('selected', `select-${moodName}`);
            badgeElement.classList.remove('animating');
            void badgeElement.offsetWidth; 
            badgeElement.classList.add('animating');

            const now = new Date();
            let hours = now.getHours();
            const minutes = String(now.getMinutes()).padStart(2, '0');
            const ampm = hours >= 12 ? 'PM' : 'AM';
            hours = hours % 12 || 12;
            const timeStr = `${hours}:${minutes} ${ampm}`;

            const activeTodayDate = now.getDate();
            
            if (selectedCalendarDayId === activeTodayDate && !hasLoggedTodayCheckIn) {
                runningStreakCount++;
                hasLoggedTodayCheckIn = true;
                updateStreakUI();
            }

            historyMoodDatabase[selectedCalendarDayId] = { type: moodName, img: moodImgMap[moodName] };
            buildCalendarTimelineDeck(); 

            moodDataset.push({ type: moodName, time: timeStr });
            
            const stream = document.getElementById('analytics-pulse-stream');
            if (stream) {
                const newStreamCard = document.createElement('div');
                newStreamCard.className = 'feed-item';
                newStreamCard.style.borderLeftColor = 'var(--pastel-pink)';
                newStreamCard.innerHTML = `
                    <div class="feed-meta" style="color:var(--dark-text)">Tracked Day Slot ${selectedCalendarDayId} • ${timeStr}</div>
                    <div class="feed-text"><img src="${moodImgMap[moodName]}" alt=""> Saved data entry profile cleanly into target status: <strong>${moodName}</strong>.</div>
                `;
                stream.insertBefore(newStreamCard, stream.firstChild);
            }
            calculateMetrics();

            const streakCard = document.getElementById('streak-card');
            if (streakCard) {
                streakCard.style.transform = 'scale(1.08) rotate(2deg)';
                setTimeout(() => streakCard.style.transform = 'scale(1) rotate(0deg)', 220);
            }
        }

        function calculateMetrics() {
            const total = moodDataset.length;
            if (total === 0) return;

            let counts = { Happy: 0, Cozy: 0, Tired: 0, Awful: 0 };
            moodDataset.forEach(entry => { counts[entry.type]++; });

            let maxCount = -1;
            let dominantVibe = 'Cozy';

            Object.keys(counts).forEach(key => {
                const percentage = Math.round((counts[key] / total) * 100);
                const pctEl = document.getElementById(`pct-${key}`);
                const fillEl = document.getElementById(`fill-${key}`);
                if (pctEl) pctEl.innerText = `${percentage}%`;
                if (fillEl) fillEl.style.width = `${percentage}%`;

                if (counts[key] > maxCount) {
                    maxCount = counts[key];
                    dominantVibe = key;
                }
            });
            
            const domDisplay = document.getElementById('dominant-vibe-display');
            if (domDisplay) {
                if (total > 0) {
                    domDisplay.innerHTML = `<img src="${moodImgMap[dominantVibe]}" alt=""> ${dominantVibe}`;
                } else {
                    domDisplay.innerHTML = `None ✨`;
                }
            }
        }

        function handleNextSlide() {
            if (currentSlide < totalSlides) {
                changeSlide(currentSlide + 1);
            } else {
                changeSlide(1);
                navigateTo('screen-login');
            }
        }

        function changeSlide(targetSlide) {
            document.getElementById(`slide-${currentSlide}`).classList.remove('active');
            document.getElementById(`dot-${currentSlide}`).classList.remove('active');
            currentSlide = targetSlide;
            document.getElementById(`slide-${currentSlide}`).classList.add('active');
            document.getElementById(`dot-${currentSlide}`).classList.add('active');
            document.getElementById('next-slide-btn').innerText = (currentSlide === totalSlides) ? '✓' : '➔';
        }

        let touchStartX = 0;
        let touchEndX = 0;
        const swipeZone = document.getElementById('swipe-zone');

        swipeZone.addEventListener('touchstart', e => { touchStartX = e.changedTouches[0].screenX; }, { passive: true });
        swipeZone.addEventListener('touchend', e => { touchEndX = e.changedTouches[0].screenX; handleSwipeGesture(); }, { passive: true });

        function handleSwipeGesture() {
            if (touchStartX - touchEndX > 40) {
                if (currentSlide < totalSlides) changeSlide(currentSlide + 1);
                else navigateTo('screen-login');
            } else if (touchEndX - touchStartX > 40) {
                if (currentSlide > 1) changeSlide(currentSlide - 1);
            }
        }

        function navigateTo(screenId) {
            document.querySelectorAll('.app-screen').forEach(screen => screen.classList.remove('active'));
            document.getElementById(screenId).classList.add('active');
            document.querySelectorAll('.nav-item').forEach(item => item.classList.remove('active'));
            
            const currentHomeScroll = document.querySelector('.home-scroll-container');
            if (currentHomeScroll) currentHomeScroll.scrollTop = 0;
            
            const currentAnalyticsScroll = document.querySelector('.analytics-scroll-container');
            if (currentAnalyticsScroll) currentAnalyticsScroll.scrollTop = 0;

            if (screenId === 'screen-home') {
                document.querySelectorAll('.bottom-nav-bar').forEach(bar => bar.querySelectorAll('.nav-item')[0].classList.add('active'));
                buildCalendarTimelineDeck(); 
            } else if (screenId === 'screen-analytics') {
                document.querySelectorAll('.bottom-nav-bar').forEach(bar => bar.querySelectorAll('.nav-item')[1].classList.add('active'));
            } else if (screenId === 'screen-login') {
                document.querySelectorAll('.bottom-nav-bar').forEach(bar => bar.querySelectorAll('.nav-item')[2].classList.add('active'));
            }
        }

        function saveJournalEntry() {
            const txtArea = document.getElementById('journal-textarea');
            const entryText = txtArea.value.trim();
            if (!entryText) return;

            const now = new Date();
            let hours = now.getHours();
            const minutes = String(now.getMinutes()).padStart(2, '0');
            const ampm = hours >= 12 ? 'PM' : 'AM';
            hours = hours % 12 || 12;
            const currentTimestamp = `${hours}:${minutes} ${ampm}`;

            const feedContainer = document.getElementById('live-journal-feed');
            if (feedContainer) {
                const newCard = document.createElement('div');
                newCard.className = 'feed-item';
                newCard.innerHTML = `
                    <div class="feed-meta" style="color: var(--branding-pink);">Saved Entry (Day ${selectedCalendarDayId}) • ${currentTimestamp}</div>
                    <div class="feed-text">${escapeHtml(entryText)}</div>
                `;
                feedContainer.insertBefore(newCard, feedContainer.firstChild);
            }
            txtArea.value = "";
            txtArea.blur(); 
        }
